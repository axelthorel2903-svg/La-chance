// Renvoie le résultat (perdant / rare / gagnant) tiré par le webhook pour une // session de paiement donnée. Le navigateur interroge cet endpoint après avoir // été redirigé depuis Stripe, avec le session_id fourni dans l'URL de retour. const { createClient } = require('redis'); let client; async function getClient(){ if(!client){ client = createClient({ url: process.env.REDIS_URL }); client.on('error', (err) => console.error('Erreur Redis :', err)); } if(!client.isOpen){ await client.connect(); } return client; } module.exports = async function handler(req, res) { const sessionId = req.query.session_id; if(!sessionId){ return res.status(400).json({ error: 'session_id manquant' }); } try { const redis = await getClient(); const tier = await redis.get('result:' + sessionId); if(tier === null || tier === undefined){ // Le webhook n'a peut-être pas encore fini de traiter l'événement. return res.status(404).json({ error: 'résultat pas encore disponible' }); } res.status(200).json({ tier }); } catch (err) { res.status(500).json({ error: err.message }); } };
